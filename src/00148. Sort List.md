/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode cur = head.next;
        head.next = null;
        ListNode lHead = null, lCur = null;
        ListNode rHead = null, rCur = null;
        while (cur != null) {
            if (cur.val < head.val) {
                if (lHead == null) {
                    lHead = cur;
                    lCur = cur;
                } else {
                    lCur.next = cur;
                    lCur = cur;
                }
            } else {
                if (rHead == null) {
                    rHead = cur;
                    rCur = cur;
                } else {
                    rCur.next = cur;
                    rCur = cur;
                }
            }
            cur = cur.next;
        }
        if (lCur != null) {
            lCur.next = null;
        }
        if (rCur != null) {
            rCur.next = null;
        }
        ListNode left = sortList(lHead);
        head.next = sortList(rHead);
        if (left == null) {
            return head;
        }
        ListNode p = left;
        while (p.next != null) {
            p = p.next;
        }
        p.next = head;
        return left;
    }
}
