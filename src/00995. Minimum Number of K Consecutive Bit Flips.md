hard的题，居然用贪心就能过
```java
class Solution {
    public int minKBitFlips(int[] A, int K) {
        int ans = 0;
        for (int i = 0; i <= A.length - K; i++) {
            if (A[i] == 0) {
                ans += 1;
                for (int j = i; j < i + K; j++) {
                    A[j] ^= 1;
                }
            }
        }
        for (int i = A.length - K - 1; i < A.length; i++) {
            if (i < 0) {
                break;
            }
            if (A[i] == 0) {
                return -1;
            }
        }
        return ans;
    }
}
```
